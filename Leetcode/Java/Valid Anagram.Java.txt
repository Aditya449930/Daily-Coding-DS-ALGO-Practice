class Solution {
    public boolean isAnagram(String s, String t) {
        char[] s1 = s.toCharArray();
        char[] s2 = t.toCharArray();
        int[] ascii1 = new int[s.length()];
        int[] ascii2 = new int[t.length()];

        for(int i = 0; i < s.length(); i++){
            ascii1[i] = s1[i];
        }

        for(int i = 0; i < t.length(); i++){
            ascii2[i] = s2[i];
        }

        Arrays.sort(ascii1);
        Arrays.sort(ascii2);

        if(Arrays.equals(ascii1,ascii2))
            return true;
        else
            return false;
    } 
}
